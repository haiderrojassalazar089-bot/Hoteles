Dataset: Hotel Bookings

El dataset contiene 119,390 reservas hoteleras entre 2015 y 2017, distribuidas en dos tipos de establecimiento: City Hotel y Resort Hotel. El EDA evalúa la calidad de los datos, detecta anomalías y comprende las distribuciones clave antes del pipeline de limpieza.

  • Variables numéricas: lead_time, adr, stays_in_week_nights, adults, children…
  • Variables categóricas: hotel, meal, country, market_segment, customer_type
  • Variable objetivo: is_canceled — 0 = no cancelada · 1 = cancelada
  • 32 variables originales → 35 tras feature engineering

def limpieza_avanzada(df: pd.DataFrame) -> pd.DataFrame:
    df = df.copy()

    # 1. Duplicados
    df = df.drop_duplicates()              # -31,994 registros

    # 2. Reservas sin huéspedes
    df = df[(df['adults'] + df['children'].fillna(0) + df['babies']) > 0]

    # 3. Imputación de nulos
    df['children']   = df['children'].fillna(df['children'].median())
    df['country']    = df['country'].fillna("Unknown")
    df['has_agent']   = df['agent'].notna().astype(int)
    df['has_company'] = df['company'].notna().astype(int)
    df = df.drop(columns=['agent', 'company'])

    # 4. Feature engineering
    df['total_guests'] = df['adults'] + df['children'] + df['babies']
    df['total_nights'] = df['stays_in_weekend_nights'] + df['stays_in_week_nights']
    df['is_family']    = ((df['children'] + df['babies']) > 0).astype(int)
    df['high_season']  = df['arrival_date_month'].isin(
        ['June', 'July', 'August']).astype(int)

    # 5. Outliers — clipping IQR
    for col in ['adr', 'lead_time', 'stays_in_weekend_nights', 'stays_in_week_nights']:
        Q1, Q3 = df[col].quantile([0.25, 0.75])
        IQR = Q3 - Q1
        df[col] = df[col].clip(Q1 - 1.5*IQR, Q3 + 1.5*IQR)

    # 6. Data leakage
    if 'reservation_status_date' in df.columns:
        df = df.drop(columns=['reservation_status_date'])

    # 7. Optimización de tipos
    categoricas = ['hotel', 'meal', 'market_segment',
                    'deposit_type', 'customer_type']
    for col in categoricas:
        if col in df.columns:
            df[col] = df[col].astype('category')

    return df  # → 87,230 registros · 35 variables

Validación con Pydantic v2

schemas.py
markdown Contratos formales de datos — Input & Output ▾

¿Por qué contratos formales?

Antes de exponer datos a un modelo predictivo o a través de una API se necesitan garantías estructurales. Pydantic v2 permite definirlas de forma declarativa.

  • StrictInt / StrictFloat — rechazan coerciones silenciosas ("42" no es un int)
  • Enums formales — HotelType, Month, Label — solo valores predefinidos
  • Field constraints — rangos realistas declarados en el modelo (ge=0, le=1000)
  • @model_validator — reglas de negocio cruzadas: ≥1 huésped · ≥1 noche
  • extra="forbid" — ningún campo no definido supera el contrato
  • validate_assignment=True — revalidación ante cualquier modificación
código schemas.py — Enums, HotelBookingInput & Output ▾ 
from pydantic import BaseModel, Field, StrictInt, StrictFloat
from pydantic import model_validator, ConfigDict
from enum    import Enum

class HotelType(str, Enum):
    RESORT = "Resort Hotel"
    CITY   = "City Hotel"

class Month(str, Enum):
    JAN="January"; FEB="February"; MAR="March";  APR="April"
    MAY="May";     JUN="June";     JUL="July";   AUG="August"
    SEP="September"; OCT="October"; NOV="November"; DEC="December"

class Label(str, Enum):
    NOT_CANCELED = "Not Canceled"
    CANCELED     = "Canceled"

class HotelBookingInput(BaseModel):
    model_config = ConfigDict(extra="forbid", validate_assignment=True)

    hotel                   : HotelType
    lead_time               : StrictInt   = Field(..., ge=0,  le=1000)
    arrival_date_month      : Month
    stays_in_weekend_nights : StrictInt   = Field(..., ge=0,  le=30)
    stays_in_week_nights    : StrictInt   = Field(..., ge=0,  le=60)
    adults                  : StrictInt   = Field(..., ge=0,  le=10)
    children                : StrictFloat = Field(0,   ge=0,  le=10)
    babies                  : StrictInt   = Field(..., ge=0,  le=5)
    adr                     : StrictFloat = Field(..., ge=0,  le=5000)

    @model_validator(mode="after")
    def validar_reglas_negocio(self):
        guests = self.adults + (self.children or 0) + self.babies
        nights = self.stays_in_weekend_nights + self.stays_in_week_nights
        if guests <= 0: raise ValueError("≥ 1 huésped requerido")
        if nights <= 0: raise ValueError("≥ 1 noche requerida")
        return self

class HotelBookingOutput(BaseModel):
    model_config = ConfigDict(extra="forbid")
    prediction  : StrictInt   = Field(..., ge=0, le=1)
    probability : StrictFloat = Field(..., ge=0.0, le=1.0)
    label       : Label

    @model_validator(mode="after")
    def consistencia_prediccion(self):
        if self.prediction == 1 and self.label != Label.CANCELED:
            raise ValueError("Inconsistencia entre prediction y label")
        if self.prediction == 0 and self.label != Label.NOT_CANCELED:
            raise ValueError("Inconsistencia entre prediction y label")
        return self

import pytest
from schemas  import HotelBookingInput
from pydantic import ValidationError

# ✓ Acepta datos válidos
HotelBookingInput(
    hotel="City Hotel", lead_time=42, arrival_date_month="July",
    stays_in_weekend_nights=1, stays_in_week_nights=3,
    adults=2, children=0.0, babies=0, adr=95.5, ...
)

# ✗ Rechaza tipo incorrecto (StrictInt)
with pytest.raises(ValidationError):
    HotelBookingInput(..., lead_time="cuarenta")

# ✗ Rechaza regla de negocio: 0 huéspedes
with pytest.raises(ValidationError):
    HotelBookingInput(..., adults=0, children=0.0, babies=0)

# ✗ Rechaza campo extra no definido
with pytest.raises(ValidationError):
    HotelBookingInput(..., campo_extra="no permitido")

código main.py — orquestador FastAPI ▾ 
from fastapi           import FastAPI, HTTPException
from fastapi.responses import HTMLResponse
from pydantic          import BaseModel
import os, pandas as pd

app = FastAPI(
    title="Hotel Bookings API",
    description="Pipeline de EDA y limpieza de datos hoteleros",
    version="1.0.0"
)

OUTPUT_PATH = "data/hotel_bookings_limpio.csv"

@app.post("/limpieza", tags=["Pipeline"])
def ejecutar_limpieza():
    import eda
    from limpieza import limpieza_avanzada
    eda.main()
    df        = pd.read_csv("data/hotel_bookings.csv")
    df_limpio = limpieza_avanzada(df)
    df_limpio.to_csv(OUTPUT_PATH, index=False)
    return {"status": "ok", "output": OUTPUT_PATH}

@app.get("/limpieza", tags=["Pipeline"])
def obtener_resultado():
    if not os.path.exists(OUTPUT_PATH):
        raise HTTPException(status_code=404,
            detail="Ejecute primero POST /limpieza")
    return {"status": "ok", "output": OUTPUT_PATH}
